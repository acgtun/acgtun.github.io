--- layout: post title: Trapping Rain Water date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: int trap(vector& height) { if(height.size() < 2) return 0; vector left(height.size(), 0); vector right(height.size(), 0); for(int i = 1;i < height.size();++i) { left[i] = max(left[i - 1], height[i - 1]); } int ret = 0, r = 0; for(int i = height.size() - 2;i >= 0;--i) { r = max(r, height[i + 1]); int h = min(left[i], r); if(height[i] < h) ret += (h - height[i]); } return ret; } }; {% endhighlight %} { % endraw %}
