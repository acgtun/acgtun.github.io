--- layout: post title: Average of Levels in Binary Tree date: 2017-07-09 05:32:57 categories: leetcode --- { % raw %} {% highlight java %} /** * Definition for a binary tree node. * public class TreeNode { * int val; * TreeNode left; * TreeNode right; * TreeNode(int x) { val = x; } * } */ public class Solution { public List averageOfLevels(TreeNode root) { List ret = new ArrayList<>(); if(root == null) { return ret; } Queue q = new LinkedList<>(); q.add(root); while(!q.isEmpty()) { int s = q.size(); double sum = 0; for(int i = 0;i < s;++i) { TreeNode f = q.poll(); sum += (double) f.val; if(f.left != null) q.add(f.left); if(f.right != null) q.add(f.right); } ret.add((double) sum / (double) s); } return ret; } } {% endhighlight %} { % endraw %}
