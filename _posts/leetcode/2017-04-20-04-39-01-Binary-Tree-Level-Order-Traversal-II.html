--- layout: post title: Binary Tree Level Order Traversal II date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} /** * Definition for a binary tree node. * struct TreeNode { * int val; * TreeNode *left; * TreeNode *right; * TreeNode(int x) : val(x), left(NULL), right(NULL) {} * }; */ class Solution { public: vector> levelOrderBottom(TreeNode* root) { vector > ret; if(root == NULL) return ret; queue q; q.push(root); while(!q.empty()) { int n = q.size(); vector lev; for(int i = 0;i < n;++i) { TreeNode* front = q.front(); q.pop(); lev.push_back(front->val); if(front->left != NULL) { q.push(front->left); } if(front->right != NULL) { q.push(front->right); } } ret.push_back(lev); } reverse(ret.begin(), ret.end()); return ret; } }; {% endhighlight %} { % endraw %}
