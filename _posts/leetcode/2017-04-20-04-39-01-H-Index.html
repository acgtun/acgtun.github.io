--- layout: post title: H-Index date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: int hIndex(vector& citations) { sort(citations.begin(), citations.end(), greater()); for(int i = 0;i < citations.size();++i) { if(i >= citations[i]) return i; } return citations.size(); } }; {% endhighlight %} { % endraw %}
