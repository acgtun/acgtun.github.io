--- layout: post title: Fizz Buzz date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight java %} public class Solution { public List fizzBuzz(int n) { List res = new ArrayList<>(); for(int i = 1;i <= n;++i) { if(i % 15 == 0) { res.add("FizzBuzz"); } else if(i % 3 == 0) { res.add("Fizz"); } else if(i % 5 == 0) { res.add("Buzz"); } else { res.add(String.valueOf(i)); } } return res; } } {% endhighlight %} { % endraw %}
