--- layout: post title: Maximum Product Subarray date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: int maxProduct(vector& nums) { if(nums.size() == 0) return 0; int large = nums[0]; int small = nums[0]; int ret = large; for(int i = 1;i < nums.size();++i) { int l = nums[i] * large; int r = nums[i] * small; large = max(max(nums[i], l), r); small = min(min(nums[i], l), r); if(large > ret) ret = large; } return ret; } }; {% endhighlight %} { % endraw %}
