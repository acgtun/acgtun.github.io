--- layout: post title: Subsets date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: vector> subsets(vector& nums) { vector > ret; vector res; ret.push_back(res); if(nums.size() == 0) return ret; sort(nums.begin(), nums.end()); for(size_t i = 0;i < nums.size();++i) { size_t end = ret.size(); for(int j = 0;j < end;++j) { vector res = ret[j]; res.push_back(nums[i]); ret.push_back(res); } } return ret; } }; {% endhighlight %} { % endraw %}
