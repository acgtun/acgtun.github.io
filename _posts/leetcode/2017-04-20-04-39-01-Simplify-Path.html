--- layout: post title: Simplify Path date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: string simplifyPath(string path) { stack s; int i = 0; while(i < path.size()) { if(path[i] == '/') { while(i < path.size() && path[i] == '/')i++; int st = i; cout << st << " " << i << endl; while(i < path.size() && path[i] != '/') i++; if(i == st) break; string p = path.substr(st, i - st); cout << "p = " << p << endl; if(p == ".") continue; else if(p == ".."){ if(!s.empty()) s.pop(); } else s.push(p); } } vector v; while (!s.empty()) { v.push_back(s.top()); s.pop(); } if(v.size() == 0) return "/"; string ret; reverse(v.begin(), v.end()); for(size_t i = 0;i < v.size();++i) { ret += "/"; ret += v[i]; } return ret; } }; {% endhighlight %} { % endraw %}
