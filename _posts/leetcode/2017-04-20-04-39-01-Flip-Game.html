--- layout: post title: Flip Game date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: vector generatePossibleNextMoves(string s) { int n = s.size(); vector ret; if(n < 2) { return ret; } for(int i = 1;i < n;++i) { if(s[i - 1] == '+' && s[i] == '+') { string str = s; str[i] = '-'; str[i - 1] = '-'; ret.push_back(str); } } return ret; } }; {% endhighlight %} { % endraw %}
