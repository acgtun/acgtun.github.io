--- layout: post title: Factor Combinations date: 2017-10-20 04:27:18 categories: leetcode --- { % raw %} {% highlight java %} class Solution { private void dfs(int start, int num, List cur, List > sets) { if(num == 1) { if(cur.size() > 1) sets.add(new ArrayList<>(cur)); return ; } if(num < start) return ; if(num % start == 0) { cur.add(start); dfs(start, num / start, cur, sets); cur.remove(cur.size() - 1); } start = start + 1; while(start < num && num % start != 0) start++; dfs(start, num, cur, sets); } public List> getFactors(int n) { List > sets = new ArrayList<>(); dfs(2, n, new ArrayList<>(), sets); return sets; } } {% endhighlight %} { % endraw %}
