--- layout: post title: Single Number II date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: int singleNumber(vector& nums) { vector one(32, 0); for(int i = 0;i < nums.size();++i) { int p = 1; for(int j = 0;j < 32;++j) { one[j] += ((nums[i] & p) != 0); p *= 2; } } for(int j = 0;j < 32;++j) { one[j] = one[j] % 3; } int ret = 0; int p = 1; for(int j = 0;j < 32;j++) { ret += (one[j] == 1 ? p : 0); p *= 2; } return ret; } }; {% endhighlight %} { % endraw %}
