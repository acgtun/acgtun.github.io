--- layout: post title: Wiggle Sort date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight java %} public class Solution { private void swap(int[] nums, int i, int j) { int t = nums[i]; nums[i] = nums[j]; nums[j] = t; } public void wiggleSort(int[] nums) { for(int i = 0;i < nums.length - 1;++i) { if((i % 2 == 0 && nums[i] > nums[i + 1]) || (i % 2 == 1 && nums[i] < nums[i + 1])) swap(nums, i, i + 1); } } } {% endhighlight %} { % endraw %}
