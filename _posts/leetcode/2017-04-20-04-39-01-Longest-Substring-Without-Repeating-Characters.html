--- layout: post title: Longest Substring Without Repeating Characters date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight java %} public class Solution { public int lengthOfLongestSubstring(String s) { HashMap map = new HashMap<>(); int start = 0; int res = 0; for(int i = 0;i < s.length();++i) { char c = s.charAt(i); if(map.containsKey(c) && map.get(c) >= start) { res = Math.max(res, i - start); start = map.get(c) + 1; } map.put(c, i); } res = Math.max(res, s.length() - start); return res; } } {% endhighlight %} { % endraw %}
