--- layout: post title: Group Anagrams date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight java %} public class Solution { private String sortString(String s) { char[] str = s.toCharArray(); Arrays.sort(str); return new String(str); } public List> groupAnagrams(String[] strs) { HashMap > map = new HashMap<>(); for(int i = 0;i < strs.length;++i) { String s = sortString(strs[i]); if(!map.containsKey(s)) { map.put(s, new ArrayList<>()); } map.get(s).add(strs[i]); } List > res = new ArrayList<>(); for(Map.Entry > entry: map.entrySet()) { res.add(entry.getValue()); } return res; } } {% endhighlight %} { % endraw %}
