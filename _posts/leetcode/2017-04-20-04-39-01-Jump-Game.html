--- layout: post title: Jump Game date: 2017-04-20 04:39:01 categories: leetcode --- { % raw %} {% highlight cpp %} class Solution { public: bool canJump(vector& nums) { int n = nums.size(); if(n == 0) return false; vector reach(n, 0); reach[0] = 1; int k = 1; for(int i = 0;i < n - 1;++i) { if(reach[i] == 1) { k = max(k, i + 1); while(k <= i + nums[i]) { reach[k] = 1; if(k == n - 1) return true; k++; } } } return reach[n - 1] == 1; } }; {% endhighlight %} { % endraw %}
