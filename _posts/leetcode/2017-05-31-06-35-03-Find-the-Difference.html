--- layout: post title: Find the Difference date: 2017-05-31 06:35:03 categories: leetcode --- { % raw %} {% highlight java %} public class Solution { public char findTheDifference(String s, String t) { Map mapS = new HashMap<>(); Map mapT = new HashMap<>(); for(char c: s.toCharArray()) { mapS.put(c, mapS.getOrDefault(c, 0) + 1); } for(char c: t.toCharArray()) { if(!mapS.containsKey(c)) { return c; } mapT.put(c, mapT.getOrDefault(c, 0) + 1); } for(Map.Entry entry: mapS.entrySet()) { if(entry.getValue() + 1 == mapT.get(entry.getKey())) { return entry.getKey(); } } return '-'; } } {% endhighlight %} { % endraw %}
